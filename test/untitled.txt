"from kivy.app import App\n",
    "from kivymd.uix.screen import Screen\n",
    "from kivymd.uix.datatables import MDDataTable\n",
    "from kivy.metrics import dp\n",
    "\n",
    "class MyApplicationApp(App):\n",
    "    def build(self):\n",
    "        screen = Screen()\n",
    "        table = MDDataTable(column_data=[\n",
    "            (\"Food\",dp(30))\n",
    "        ])\n",
    "        screen.add_widget(table)\n",
    "        \n",
    "        return screen\n",
    "\n",
    "if __name__ == '__main__':\n",
    "    \n",
    "    x=MyApplicationApp()\n",
    "    x.run()\n",
    "    "
